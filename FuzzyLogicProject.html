# We need the activation of our fuzzy membership functions at these values. 
# The exact values 41 and 11.8 do not exist on our universes... 
# This is what fuzz.interp_membership exists for!

eggsize_fit_small = fuzz.interp_membership(x_eggsize, eggsize_small, input_eggsize)
eggsize_fit_medium = fuzz.interp_membership(x_eggsize, eggsize_medium, input_eggsize)
eggsize_fit_big = fuzz.interp_membership(x_eggsize, eggsize_big, input_eggsize)
